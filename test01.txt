The minimial program.


    Main : Output
      IO.With(F)

    F() : ()
      IO.Print(“Hi”)
      IO.Print(“Bye”)

    mod IO

      new E = Escape⟦String, ()⟧

      With(f : () ⟶ ()) : Output
        Loop(E.Catch(f))

      Loop(s : Stream⟦String, (), ()⟧) : Output
        case s
          End()
            Exit
          Next(x)(k)
            Write(x)(Continue(Compose(Loop)(k)))

      Print(x : String) : ()
        E.Throw(x)

    Constant⟦a, b⟧(x : a)(_ : b) : a
      Identity⟦a⟧(x)

    Identity⟦a⟧(x : a) : a
      x

    Compose⟦a, b, c⟧(f : b ⟶ c)(g : a ⟶ b)(x : a) : c
      f(g(x))
