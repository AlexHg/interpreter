The minimial program.


    Main : Output
      IO.With(F)

    F() : Output
      IO.Print("Hello")
      IO.Print("Bye")
      Exit

    mod IO

      new E = Escape⟦String, ()⟧

      With(f : () ⟶ Output) : Output
        for (x)(k) in E.Catch(f)
          Write(x)(Continue(k))

      Print(x : String) : ()
        E.Throw(x)

    Constant⟦a, b⟧(x : a)(_ : b) : a
      Identity⟦a⟧(x)

    Identity⟦a⟧(x : a) : a
      x

    Written(x : String)(f : () ⟶ Output) : Output
      Write(x)(Continue(f))
